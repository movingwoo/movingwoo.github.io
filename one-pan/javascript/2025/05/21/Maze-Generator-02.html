function addFrontiers(x, y, frontiers, frontierSet) {
    const dirs = [[0, -2], [2, 0], [0, 2], [-2, 0]];

    for (let [dx, dy] of dirs) {
        const nx = x + dx;
        const ny = y + dy;

function findSet(sets, key) {
    if (sets.get(key) !== key) {

function union(sets, set1, set2) {
