function shuffle(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}
  
// ÎØ∏Î°ú ÏÉùÏÑ±(DFS)
async function generateDFS(x = entrance[0], y = entrance[1]) {
  const key = `${x},${y}`;
  visited.add(key);
  maze[y][x] = 0;
  drawCell(x, y, 'white'); // Ìù∞ÏÉâ Í∏∏

  // Ïù¥Îèô Î∞©Ìñ• Î¨¥ÏûëÏúÑ
  const dirs = shuffle([[0, -2], [2, 0], [0, 2], [-2, 0]]);

  // forEach ÏÇ¨Ïö©Ïãú ÎèôÍ∏∞Ìôî Î¨∏Ï†úÏûàÏñ¥ÏÑú for of ÏÇ¨Ïö©
  for (let [dx, dy] of dirs) {
    const nx = x + dx;
    const ny = y + dy;
    const betweenX = x + dx / 2;
    const betweenY = y + dy / 2;
    const neighborKey = `${nx},${ny}`;

    // Î∞©Î¨∏ÌïòÏßÄ ÏïäÏùÄ ÏÖÄÏùÑ ÌÜµÎ°úÎ°ú ÎßåÎì§Í∏∞
    if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && !visited.has(neighborKey)) {
      maze[ny][nx] = 0; // Îã§Ïùå ÏÖÄÎèÑ ÌÜµÎ°ú
      maze[betweenY][betweenX] = 0; // Ï§ëÍ∞Ñ Î≤Ω ÏóÜÏï†Í∏∞
      drawCell(betweenX, betweenY, 'white');

      await sleep(5);
      await generateDFS(nx, ny);
    }
  }
}
  
// Í∞ÄÏû• Î©ÄÎ¶¨ Îñ®Ïñ¥ÏßÑ ÏÖÄÏùÑ Ï∂úÍµ¨Î°ú ÏßÄÏ†ï(BFS)
function findFurthestCell(sx, sy) {
  const que = [[sx, sy, 0]];
  const visited = new Set();
  let maxDist = -1;
  let furthest = [sx, sy];

  while (que.length) {
    const [x, y, dist] = que.shift();
    const key = `${x},${y}`;

    if (visited.has(key)) {
      continue;
    }

    visited.add(key);

    if (dist > maxDist) {
      maxDist = dist;
      furthest = [x, y];
    }

    for (let [dx, dy] of [[0,1],[1,0],[0,-1],[-1,0]]) {
      const nx = x + dx, ny = y + dy;
      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && maze[ny][nx] === 0) {
        que.push([nx, ny, dist + 1]);
      }
    }
  }

  return furthest;
}

function findPathBFS(sx, sy, ex, ey) {
  const que = [[sx, sy]];
  const visited = new Set();
  const parent = {};

  while (que.length) {
    const [x, y] = que.shift();

    if (x === ex && y === ey) {
      break;
    }
    const key = `${x},${y}`;

    visited.add(key);

    for (let [dx, dy] of [[0,1],[1,0],[0,-1],[-1,0]]) {
      const nx = x + dx, ny = y + dy;
      const nextKey = `${nx},${ny}`;

      // Î≤ΩÏù¥ ÏïÑÎãàÍ≥† Î∞©Î¨∏ÌïòÏßÄ ÏïäÏùÄ Í∏∏
      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && maze[ny][nx] === 0 && !visited.has(nextKey)) {
        que.push([nx, ny]);
        visited.add(nextKey);
        parent[nextKey] = key;
      }
    }
  }

  // Ï∂úÍµ¨ÏóêÏÑú Ïó≠ÏúºÎ°ú Í≤ΩÎ°ú Ï∂îÏ†Å
  const path = [];
  let curr = `${ex},${ey}`;

  while (curr !== `${sx},${sy}`) {
    const [x, y] = curr.split(',').map(Number);
    path.push([x, y]);
    curr = parent[curr];
  }

  path.push([sx, sy]);

  return path.reverse();
}
  
// ÎØ∏Î°ú ÌÉàÏ∂ú Í≤ΩÎ°ú Í∑∏Î¶¨Í∏∞
async function drawPathAnimated(path) {
  for (let [x, y] of path) {
    if ((x === entrance[0] && y === entrance[1]) || (x === exit[0] && y === exit[1])) {
      continue;
    }

    // Îπ®Í∞ïÏúºÎ°ú ÏÉâÏπ†
    drawCell(x, y, 'red');
    await sleep(10);
  }
}
