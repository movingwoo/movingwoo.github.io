    3. мғҲлҘј к·ёл Өм„ң м җн”„мҷҖ мӨ‘л Ҙ кө¬нҳ„

    мғҲлқјкі  н•ҳкёҙ н–Ҳм§Җл§ҢвҖҰ мқјлӢЁ лҸҷк·ёлқјлҜёлҘј к·ёл ӨлҶ“кі  мғҲлқјкі  мҡ°кё°мһҗ.
    мІҳмқҢм—җ кіөмӨ‘м—җм„ң мӢңмһ‘н–ҲлӢӨк°Җ мӨ‘л Ҙмқ„ л°ӣм•„ м•„лһҳлЎң л–Ём–ҙм§„лӢӨ.
    к·ёлҰ¬кі  нҒҙлҰӯн•ҳл©ҙ кіөмӨ‘мңјлЎң нҠҖм–ҙмҳӨлҘёлӢӨ.
    м–јл§ҲлӮҳ л№ лҘҙкІҢ л–Ём–ҙм§Җкі  лҜјк°җн•ҳкІҢ нҠҖм–ҙмҳӨлҘјм§ҖлҠ” лӢӨ л§Ңл“Өкі  мЎ°м Ҳн•ҳкё°лЎң н•Ё.

    // мғҲ
const bird = {
  // мӨ‘л Ҙкіј м җн”„л Ҙмқ„ м¶”нӣ„ мЎ°м •
  x: 150,
  y: 240,
  radius: 20,
  velocity: 0,
  gravity: 0.5,
  jump: -8
};

let animationFrameId;

canvas.addEventListener('click', () => {
  // мҶҚлҸ„м—җ м җн”„л Ҙмқ„ лҚ”н•ҙ м җн”„
  bird.velocity = bird.jump;
});

// кІҢмһ„ лЈЁн”„
function run() {
  // л¬јлҰ¬ кі„мӮ°
  // мҶҚлҸ„лҠ” мӨ‘л Ҙл§ҢнҒј м¶”к°Җ, мҶҚлҸ„л§ҢнҒј yм¶• мЎ°м Ҳ
  bird.velocity += bird.gravity;
  bird.y += bird.velocity;

  // нҷ”л©ҙ м§Җмҡ°кё°
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // мғҲ к·ёлҰ¬кё°
  ctx.beginPath();
  ctx.arc(bird.x, bird.y, bird.radius, 0, Math.PI * 2);
  ctx.fillStyle = '#ffcc00';
  ctx.fill();
  ctx.strokeStyle = '#333';
  ctx.stroke();

  // мІңмһҘ л°”лӢҘ м ңн•ң
  if (bird.y + bird.radius > canvas.height) {
    bird.y = canvas.height - bird.radius;
    bird.velocity = 0;
  }
  if (bird.y - bird.radius < 0) {
    bird.y = bird.radius;
    bird.velocity = 0;
  }

  animationFrameId = requestAnimationFrame(run);
}

// кІҢмһ„ мӢңмһ‘
run();
    4. мһҘм• л¬ј кө¬нҳ„

    мҳӨлҘёмӘҪм—җм„ң мҷјмӘҪмңјлЎң мң„/м•„лһҳ л°°м№ҳлҗң мһҘм• л¬јмқҙ лӢӨк°ҖмҳЁлӢӨ.
    м җн”„лҘј нҶөн•ҙ мһҘм• л¬јмқ„ н”јн•ҳлҠ”кІҢ лӘ©м Ғмқё кІҢмһ„мқҙлӢӨ.
    мһҘм• л¬ј мң„мҷҖ м•„лһҳмқҳ к°„кІ©(н”јн•  кө¬л©Қ)мқҖ к°ҷм•„м•јн•ҳл©°, мһҘм• л¬ј нҒ¬кё°лҠ” лһңлҚӨмңјлЎң мғқм„ұн•ңлӢӨ.

    // мһҘм• л¬јмқҖ кі„мҶҚ м¶”к°Җ, м ңкұ°лҗҳкё° л•Ңл¬ём—җ мғҲмІҳлҹј кі м •к°қмІҙлЎң м„Өм •н•ҳл©ҙ м•Ҳлҗҳкі  л°°м—ҙлЎң м„Өм •
// мһҘм• л¬ј л„Ҳл№„мҷҖ мҶҚлҸ„ л“ұ м¶”нӣ„ мЎ°м •
const pipeWidth = 20;
const pipeGap = 180; // мң„ м•„лһҳ к°„кІ©
const pipeSpeed = 5;
const pipeInterval = 90; // лӢӨмқҢ мһҘм• л¬ј к°„кІ©
let frameCount = 0;
let pipes = [];

// мһҘм• л¬ј м¶”к°Җ
function addPipe() {
  // мһҘм• л¬јмқҳ мөңмҶҢ лҶ’мқҙлҘј м§Җм • нӣ„ лӮҳлЁём§Җ кі„мӮ°
  const minPipeHeight = 50;
  const maxPipeHeight = canvas.height - pipeGap - minPipeHeight;
  const topPipeHeight = Math.floor(Math.random() * (maxPipeHeight - minPipeHeight + 1)) + minPipeHeight;

  pipes.push({
    x: canvas.width,
    topHeight: topPipeHeight,
    bottomY: topPipeHeight + pipeGap
  });
}

// мһҘм• л¬ј к·ёлҰ¬кё°
function drawPipes() {
  ctx.fillStyle = '#228B22';

  for (let pipe of pipes) {
    // мң„мӘҪ мһҘм• л¬ј
    ctx.fillRect(pipe.x, 0, pipeWidth, pipe.topHeight);

    // м•„лһҳмӘҪ мһҘм• л¬ј
    ctx.fillRect(pipe.x, pipe.bottomY, pipeWidth, canvas.height - pipe.bottomY);
  }
}

// мһҘм• л¬ј к°ұмӢ 
function updatePipes() {
  for (let pipe of pipes) {
    // мһҘм• л¬ј мҶҚлҸ„м—җ л”°лқј xм¶• мЎ°м Ҳ
    pipe.x -= pipeSpeed;
  }

  // нҷ”л©ҙ л°–мңјлЎң лӮҳк°„ мһҘм• л¬ј м ңкұ°
  pipes = pipes.filter(pipe => pipe.x + pipeWidth > 0);
}
    5. м¶©лҸҢ нҢҗм • кө¬нҳ„

    мғҲмҷҖ мһҘм• л¬јмқҙ м¶©лҸҢн–Ҳмқ„ л•Ң кІҢмһ„мҳӨлІ„к°Җ лҗҳм–ҙм•јн•ңлӢӨ.
    мһҘм• л¬ј лҝҗл§Ң м•„лӢҲлқј мІңмһҘмқҙлӮҳ л°”лӢҘм—җ лӢҝм•„лҸ„ кІҢмһ„мҳӨлІ„лӢӨ.

    мӢ лӮҳлҠ” мҲҳн•ҷмӢңк°„~~
    мӣҗкіј м§ҒмӮ¬к°Ғнҳ• м¶©лҸҢ нҢҗм • кіөмӢқмқҙ мЎҙмһ¬н•ҳлҜҖлЎң мӮ¬мҡ©н•ңлӢӨ.

    // мғҲмҷҖ нҢҢмқҙн”„ мӮ¬мқҙ к°ҖмһҘ к°Җк№Ңмҡҙ м җ
// к°’мқ„ мөңмҶҢк°’ мөңлҢҖк°’ мӮ¬мқҙлЎң м ңн•ң
function clamp(value, min, max) {
  return Math.max(min, Math.min(max, value));
}

// м җмқҙ мӣҗ м•Ҳм—җ л“Өм–ҙмһҲлҠ”м§Җ нҷ•мқё
// кұ°лҰ¬кіөмӢқ нҷңмҡ© (лЈЁнҠё м—°мӮ°мқҖ м„ұлҠҘ л–Ём–ҙм ём„ң көімқҙ м“ё н•„мҡ” м—ҶмқҢ)
function circleCollides(cx, cy, cr, px, py) {
  const dx = cx - px;
  const dy = cy - py;
  return dx * dx + dy * dy <= cr * cr;
}

    кіөмӢқмқ„ мқҙмҡ©н•ҙ м¶©лҸҢ м—¬л¶ҖлҘј нҢҗлӢЁн•ҳкі  м¶©лҸҢ нҢҗм • мӢң кІҢмһ„мҳӨлІ„ мӢңнӮЁлӢӨ.
    мІңмһҘмқҙлӮҳ л°”лӢҘмқҖ к·ёлғҘ мўҢн‘ңнҷ•мқёл§Ң н•ҳл©ҙ лҒқлӮңлӢӨ.

    // мІңмһҘ л°”лӢҘ м¶©лҸҢ мІҙнҒ¬
if (bird.y + bird.radius >= canvas.height || bird.y - bird.radius <= 0) {
  gameOver();
  return;
}

// мһҘм• л¬ј м¶©лҸҢ мІҙнҒ¬
for (let pipe of pipes) {
  // мһҘм• л¬ј xм¶• лІ”мң„ лӮҙ мғҲмҷҖ к°ҖмһҘ к°Җк№Ңмҡҙ x мўҢн‘ң
  const closestX = clamp(bird.x, pipe.x, pipe.x + pipeWidth);

  // мң„ мһҘм• л¬јкіј мғҲмқҳ yмўҢн‘ң л№„көҗ
  const topY = clamp(bird.y, 0, pipe.topHeight);

  // м¶©лҸҢн•ҳлҠ”м§Җ нҷ•мқё
  if (circleCollides(bird.x, bird.y, bird.radius, closestX, topY)) {
      gameOver();
      return;
  }

  // м•„лһҳ мһҘм• л¬јкіј мғҲмқҳ yмўҢн‘ң л№„көҗ
  const bottomY = clamp(bird.y, pipe.bottomY, canvas.height);

  // м¶©лҸҢн•ҳлҠ”м§Җ нҷ•мқё
  if (circleCollides(bird.x, bird.y, bird.radius, closestX, bottomY)) {
      gameOver();
      return;
  }
}

// кІҢмһ„мҳӨлІ„
function gameOver() {
  cancelAnimationFrame(animationFrameId);
  ctx.fillStyle = 'red';
  ctx.font = '48px Arial';
  ctx.textAlign = 'center';
  ctx.fillText('Game Over', canvas.width / 2, canvas.height / 2);
}

// мһҘм• л¬ј мғқм„ұ мӢң м җмҲҳ нҢҗм •мҡ© ліҖмҲҳлҘј м¶”к°Җ

// м җмҲҳ к·ёлҰ¬кё°
function drawScore() {

    л°ҳм„ұ

    лӮңмқҙлҸ„ мЎ°м Ҳмқҳ м–ҙл ӨмӣҖ
    мҡ” лӢЁмҲңн•ң л”ёк№Қ кІҢмһ„ л§Ңл“ңлҠ”лҚ°лҸ„ лӮңмқҙлҸ„ мЎ°м •м—җ л§ҺмқҖ кі лҜјмқ„ н•ҳкІҢ лҗңлӢӨ.
    мӢ¬м§Җм–ҙ лӮҙк°Җ мғқк°Ғн•ҳкё°м—җ мһҘм• л¬ј мҶҚлҸ„л§ҢнҒј мғҲмқҳ мҶҚлҸ„лҸ„ кі л Өн•ҳлҠ”кІҢ л°ёлҹ°мҠӨ мғҒ л§һм§Җ м•ҠлӮҳ мӢ¶мқҖлҚ°вҖҰ
    к°“кІң м ңмһ‘мһҗл“ӨмқҖ м–јл§ҲлӮҳ л§ҺмқҖ кі лҜјмқ„ нҶөн•ҙ лӮңмқҙлҸ„лҘј м •н•ҳкІҢ лҗҳм—Ҳмқ„к№Ң?

    мӣҖм§ҒмқҙлҠ” л°°кІҪ
    л§Ңм•Ҫ л°°кІҪм—җ кө¬лҰ„к°ҷмқҖ л¬ҙлҠ¬лқјлҸ„ лқ„м–ҙл‘җкі  л°°кІҪмқ„ мһҘм• л¬ј мҶҚлҸ„м—җ л§һм¶° мҠӨнҒ¬лЎӨн–Ҳмңјл©ҙ лҚ” мғқлҸҷк°җмқҙ мғқкІјмқ„кұ°лӢӨ.
    л°”лӢҘмқҙлӮҳ мІңмһҘлҸ„ лҜјм§ңк°Җ м•„лӢҲлқј мҳӨлҸҢнҶ лҸҢн•ҳкІҢ н‘ңмӢңн•ҙл‘җкі  мҠӨнҒ¬лЎӨн•ҳл©ҙ м—„мІӯлӮң мҶҚлҸ„к°җмқ„ лҠҗлӮ„ мҲҳ мһҲкІ м§ҖвҖҰ

    мҠӨн„°н„°л§Ғ?

    мӢңк°„ кё°л°ҳ мқҙлҸҷмқ„ мӮ¬мҡ©н•ҳл©ҙ н”„л Ҳмһ„мқҙ м•Ҫк°„ л°Җл ӨлҸ„ л¶Җл“ңлҹҪкІҢ ліҙм •лҗңлӢӨ.
    лӢӨмқҢ мә”лІ„мҠӨ кІҢмһ„ л§Ңл“Өл•ҢлҠ” н”„л Ҳмһ„лҸ„ мӢ кІҪм“°л©ҙм„ң л§Ңл“Өм–ҙлҙҗм•јкІ лӢӨ.

    к·ёлһҳлҸ„ л°ҖлҰ¬м„ём»Ёл“ң лӢЁмң„лЎң мӢ кІҪ м“ё н•„мҡ”лҠ” м—ҶлӢӨкі вҖҰ
